#include <WiFi.h>
#include <HTTPClient.h>
#include <SPI.h>
#include <MFRC522.h>
#include <ArduinoJson.h>

// 1. KONFIGURASI WIFI
const char* ssid = "NAMA_WIFI_KAMU";
const char* password = "PASSWORD_WIFI_KAMU";

// 2. KONFIGURASI SERVER (RAILWAY)
// Endpoint 'scan' adalah rute publik, tidak butuh JWT. 
// Backend akan otomatis handle IN/OUT atau REGISTER sesuai 'IoT Mode' yang diset di App Android.
const char* serverName = "https://ktormachinelearning-production.up.railway.app/api/v1/inventory/scan";

// 3. PIN DEFINITIONS (ESP32)
#define SS_PIN      5   // SDA
#define RST_PIN     22  // Reset
#define BUZZER_PIN  21  // Pin Buzzer (+)

MFRC522 rfid(SS_PIN, RST_PIN);

void setup() {
  Serial.begin(115200);
  pinMode(BUZZER_PIN, OUTPUT);
  digitalWrite(BUZZER_PIN, LOW);

  // Koneksi WiFi
  WiFi.begin(ssid, password);
  Serial.print("Connecting to WiFi");
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }
  Serial.println("\nWiFi Connected!");

  // Inisialisasi SPI & RFID
  SPI.begin();
  rfid.PCD_Init();
  Serial.println("RFID Reader Ready. Scan a tag...");
}

void loop() {
  // Cek apakah ada kartu baru
  if (!rfid.PCD_IsNewCardPresent()) return;
  if (!rfid.PCD_ReadCardSerial()) return;

  // 1. Ambil UID Kartu
  String tagUid = "";
  for (byte i = 0; i < rfid.uid.size; i++) {
    tagUid += String(rfid.uid.uidByte[i] < 0x10 ? "0" : "");
    tagUid += String(rfid.uid.uidByte[i], HEX);
  }
  tagUid.toUpperCase();
  
  Serial.print("\nTag Scanned: ");
  Serial.println(tagUid);
  
  // Bunyi Buzzer (Beep)
  digitalWrite(BUZZER_PIN, HIGH);
  delay(150);
  digitalWrite(BUZZER_PIN, LOW);

  // 2. Kirim ke Server Ktor
  if (WiFi.status() == WL_CONNECTED) {
    HTTPClient http;
    http.begin(serverName);
    http.addHeader("Content-Type", "application/json");

    // Buat JSON Body
    StaticJsonDocument<200> doc;
    doc["tag_uid"] = tagUid;
    
    // Tips: Jika ingin kirim manual event_type bisa tambahkan:
    // doc["event_type"] = "OUT"; 
    // Tapi jika dikosongkan, backend akan handle otomatis (Smart Mode).

    String requestBody;
    serializeJson(doc, requestBody);

    Serial.println("Sending data to server...");
    int httpResponseCode = http.POST(requestBody);

    if (httpResponseCode > 0) {
      String response = http.getString();
      Serial.print("HTTP Response Code: ");
      Serial.println(httpResponseCode);
      Serial.println("Response: " + response);
      
      // Berhasil (Visual Feedback)
      if (httpResponseCode == 200 || httpResponseCode == 201) {
          // Double Beep
          digitalWrite(BUZZER_PIN, HIGH); delay(50); digitalWrite(BUZZER_PIN, LOW);
      }
    } else {
      Serial.print("Error sending POST: ");
      Serial.println(httpResponseCode);
    }
    http.end();
  }

  // Jeda agar tidak double scan
  rfid.PCD_StopCrypto1();
  delay(2000);
}
